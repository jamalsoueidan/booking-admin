## React Theme Context

Use this component when you want to show a modal anywhere WITHOUT link just overlay.

### Component usage

```tsx
function MockComponent() {
  const { setPrimaryAction } = useContext(ModalContext);

  useEffect(() => {
    setPrimaryAction({
      content: "TilfÃ¸j",
      onAction: () => {
        console.log("Add");
      },
    });
  }, []);

  return <>MockComponent that read value from context</>;
}

const BasicThemeUsage = () => (
  <ModalProvider title="hej med dig" open onClose={() => {}}>
    <MockComponent />
  </ModalProvider>
);
```
